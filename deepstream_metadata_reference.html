Full Metadata Hierarchy

Every GstBuffer in DeepStream carries a NvDsBatchMeta — a tree structure that accumulates metadata as the buffer passes through each plugin. The tree deepens with each stage: PGIE adds object bboxes, tracker adds IDs, each SGIE appends classifier or tensor results.

NvDsBatchMeta one per GstBuffer · attached to batch GstBuffer
NvDsFrameMeta one per source stream · frame_num, source_id, surface_ref frame_meta_list
NvDsObjectMeta one per detected object · bbox, class_id, confidence, object_id obj_meta_list
NvDsClassifierMeta top-1 label + prob from SGIE · one per SGIE that ran on this object classifier_meta_list
NvDsLabelInfo result_label (str) · result_prob (float) · label_id (int) label_info_list
NvDsUserMeta container for raw tensor output · meta_type = NVDSINFER_TENSOR_OUTPUT_META obj_user_meta_list
NvDsInferTensorMeta raw output layers · unique_id · output_layers_info[i].buffer (void*) user_meta_data
NvDsPastFrameObjBatch terminated track history · bbox + confidence per past frame past_frame_obj_batch
Key distinction classifier_meta_list is populated when output-tensor-meta=0 (default). obj_user_meta_list is populated when output-tensor-meta=1. The two are mutually exclusive per SGIE instance.

read classifier result for SGIE-2 (color)Python
def get_color_label(obj_meta, sgie_uid=2):
    l_cls = obj_meta.classifier_meta_list
    while l_cls:
        cls_meta = pyds.NvDsClassifierMeta.cast(l_cls.data)
        if cls_meta.unique_component_id == sgie_uid:
            l_lbl = cls_meta.label_info_list
            while l_lbl:
                lbl = pyds.NvDsLabelInfo.cast(l_lbl.data)
                return lbl.result_label, lbl.result_prob
                l_lbl = l_lbl.next
        l_cls = l_cls.next
    return None, 0.0

NvDsUserMeta · Tensor Output Path

When output-tensor-meta=1 is set, nvinfer wraps a NvDsInferTensorMeta inside a NvDsUserMeta and attaches it to obj_user_meta_list. The meta_type field distinguishes it from any other user metadata on the same list.

NvDsInferTensorMeta

extract raw tensor — genericPython · ctypes
import ctypes, numpy as np

def extract_tensor(obj_meta, sgie_uid: int, n_classes: int):
    l_user = obj_meta.obj_user_meta_list
    while l_user:
        user_meta = pyds.NvDsUserMeta.cast(l_user.data)
        if user_meta.base_meta.meta_type != \
                pyds.NvDsMetaType.NVDSINFER_TENSOR_OUTPUT_META:
            l_user = l_user.next; continue

        tmeta = pyds.NvDsInferTensorMeta.cast(user_meta.user_meta_data)
        if tmeta.unique_id != sgie_uid:
            l_user = l_user.next; continue

        for i in range(tmeta.num_output_layers):
            layer = pyds.get_nvds_LayerInfo(tmeta, i)
            total = 1
            for d in range(layer.dims.numDims):
                total *= layer.dims.d[d]
            if total != n_classes: continue

            ptr   = ctypes.cast(layer.buffer,
                                ctypes.POINTER(ctypes.c_float))
            probs = np.ctypeslib.as_array(ptr, shape=(n_classes,)).copy()
            return probs   # .copy() is non-negotiable
        l_user = l_user.next
    return None
Buffer Lifetime layer.buffer is owned by nvinfer and freed immediately after the probe callback returns. Always call .copy(). Never store the raw ctypes pointer or np.ctypeslib.as_array view without copying.

Full Probe — Reading All Three SGIEs

multi_sgie_probe.pyPython · pyds
import pyds, ctypes, numpy as np
from collections import defaultdict

# ── SGIE registry ────────────────────────────────────────────────
COLOR_SGIE_UID  = 2   # color classifier,  15 classes
LP_DET_SGIE_UID = 3   # LP detector,        network-type=0
LP_CLS_SGIE_UID = 4   # LP char classifier, network-type=1

N_COLOR_CLASSES = 15
LP_CHARS        = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"
N_LP_CHARS      = len(LP_CHARS)   # 36

COLOR_LABELS = [
    "black", "white", "silver", "gray", "red", "blue",
    "green", "yellow", "orange", "brown", "purple",
    "beige", "dark_blue", "dark_green", "maroon"
]

# ── Tensor extraction (generic) ───────────────────────────────────
def extract_tensor(obj_meta, uid, n):
    l = obj_meta.obj_user_meta_list
    while l:
        um = pyds.NvDsUserMeta.cast(l.data)
        if um.base_meta.meta_type == \
                pyds.NvDsMetaType.NVDSINFER_TENSOR_OUTPUT_META:
            tm = pyds.NvDsInferTensorMeta.cast(um.user_meta_data)
            if tm.unique_id == uid:
                for i in range(tm.num_output_layers):
                    layer = pyds.get_nvds_LayerInfo(tm, i)
                    total = 1
                    for d in range(layer.dims.numDims):
                        total *= layer.dims.d[d]
                    if total == n:
                        ptr = ctypes.cast(layer.buffer,
                              ctypes.POINTER(ctypes.c_float))
                        return np.ctypeslib.as_array(
                                   ptr, shape=(n,)).copy()
        l = l.next
    return None

# ── Label-mode reader (NvDsClassifierMeta) ───────────────────────
def get_label(obj_meta, uid):
    l = obj_meta.classifier_meta_list
    while l:
        cm = pyds.NvDsClassifierMeta.cast(l.data)
        if cm.unique_component_id == uid:
            ll = cm.label_info_list
            while ll:
                li = pyds.NvDsLabelInfo.cast(ll.data)
                return li.result_label, li.result_prob
                ll = ll.next
        l = l.next
    return None, 0.0

# ── Main probe ────────────────────────────────────────────────────
def multi_sgie_probe(pad, info, u_data):
    gst_buffer = info.get_buffer()
    batch_meta  = pyds.gst_buffer_get_nvds_batch_meta(hash(gst_buffer))

    l_frame = batch_meta.frame_meta_list
    while l_frame:
        frame_meta = pyds.NvDsFrameMeta.cast(l_frame.data)

        l_obj = frame_meta.obj_meta_list
        while l_obj:
            obj  = pyds.NvDsObjectMeta.cast(l_obj.data)
            rect = obj.rect_params
            key  = (frame_meta.source_id, obj.object_id)

            result = {
                "frame_num":   frame_meta.frame_num,
                "source_id":   frame_meta.source_id,
                "object_id":   obj.object_id,
                "class_id":    obj.class_id,
                "pgie_conf":   obj.confidence,
                "tracker_conf":obj.tracker_confidence,
                "bbox": (rect.left, rect.top,
                          rect.width, rect.height),
            }

            # ── SGIE-2: Color classifier (label mode) ────────
            color_lbl, color_prob = get_label(obj, COLOR_SGIE_UID)
            result["color_label"] = color_lbl
            result["color_prob"]  = color_prob

            # ── SGIE-2: Color classifier (tensor mode) ───────
            color_tensor = extract_tensor(
                obj, COLOR_SGIE_UID, N_COLOR_CLASSES
            )
            result["color_probs"] = color_tensor  # float32[15]

            # ── SGIE-3: LP detector (tensor mode) ────────────
            # LP detector exposes raw detection tensor
            lp_tensor = extract_tensor(
                obj, LP_DET_SGIE_UID, None  # variable shape
            )
            result["lp_detected"] = lp_tensor is not None

            # ── SGIE-4: LP char classifier (label mode) ──────
            lp_text, lp_conf = get_label(obj, LP_CLS_SGIE_UID)
            result["lp_text"] = lp_text
            result["lp_conf"] = lp_conf

            process_result(key, result)
            l_obj = l_obj.next
        l_frame = l_frame.next
    return Gst.PadProbeReturn.OK
